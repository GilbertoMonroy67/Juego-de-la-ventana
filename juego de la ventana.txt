void escena1() {
    int op;
    cout << "11:00 p.m. - Escena 1\n";
    cout << "El silencio es perturbador. A las 11:30 p.m., escuchas un rasguño en la ventana.\n\n";

    cout << "Opciones:\n";
    cout << "1. Mantenerte inmóvil.\n";
    cout << "2. Taparte los oídos.\n";
    cout << "3. Gritar o reaccionar con miedo.\n";
    cout << "Elige una opción: ";
    cin >> op;

    switch(op) {
        case 1: cout << "Permanece en silencio. El rasguño se detiene...\n\n"; break;
        case 2: cout << "Te tapas los oídos. Aun así, sientes el rasguño...\n\n"; break;
        case 3: cout << "Gritas. El rasguño se detiene por un momento...\n\n"; break;
        default: cout << "Opción no válida.\n\n"; break;
    }
}
